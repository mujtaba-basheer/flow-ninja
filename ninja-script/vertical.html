<script src="https://unpkg.com/swiper/swiper-bundle.min.js"></script><script>
var swiper = new Swiper('.vertical-swiper', {
direction: 'vertical',
slidesPerView: 1,
spaceBetween: 50,
grabCursor: true,
a11y: false,
autoHeight: true,
loop: true,
pagination: {
el: '.vertical-slider-pagination',
clickable: true,
},
breakpoints: {
0: {
allowTouchMove: false,
},
767:{
allowTouchMove: true,
},
988:{
allowTouchMove: true,
},
}
});
</script>
