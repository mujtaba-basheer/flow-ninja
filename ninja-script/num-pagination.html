Paste this before <body> in project settings

<script src="https://unpkg.com/swiper/swiper-bundle.min.js"></script> <!-- Initialize Swiper -->
<script>
var swiper = new Swiper("#snp-swiper", {
a11y: false,
grabCursor:true,
navigation:{
nextEl: "#right-button",
prevEl: "#left-button",
},
pagination: {
el: ".snp-pagination",
clickable: true,
/*Return bullets as numbers*/
renderBullet: function (index, className) {
return '<span class="' + className + '">' + (index + 1) + "</span>";
},
},
breakpoints: {
0: {
slidesPerView: 1.1,
spaceBetween: 20,
},
767: {
slidesPerView: 2.5,
spaceBetween: 24,
},
988: {
slidesPerView: 3,
spaceBetween: 48,
},
},
});
</script>
